<script>
// Canvas for stars/moon
const canvas=document.getElementById('bgCanvas');
const ctx=canvas.getContext('2d');
canvas.width=window.innerWidth;
canvas.height=window.innerHeight;
let stars=[];
for(let i=0;i<100;i++){
    stars.push({x:Math.random()*canvas.width,y:Math.random()*canvas.height,r:Math.random()*2+1});
}
function drawStars(){
    ctx.clearRect(0,0,canvas.width,canvas.height);
    ctx.fillStyle="white";
    stars.forEach(s=>{ctx.beginPath();ctx.arc(s.x,s.y,s.r,0,Math.PI*2);ctx.fill();});
    requestAnimationFrame(drawStars);
}
drawStars();

// Mini Game
let player, game, scoreBoard, score=0, salamis=[], gameInterval;
function startGame(){
    document.getElementById('home').classList.add('hidden');
    document.getElementById('game').classList.remove('hidden');
    player=document.getElementById('player');
    game=document.getElementById('game');
    scoreBoard=document.getElementById('scoreboard');
    for(let i=0;i<8;i++){
        let s=document.createElement('div');
        s.classList.add('salami');
        s.style.left=Math.random()*(window.innerWidth-30)+'px';
        s.style.top=Math.random()*300+'px';
        game.appendChild(s);
        salamis.push(s);
    }
    document.addEventListener('keydown',movePlayer);
    let touchStartX=0,touchStartY=0;
    document.addEventListener('touchstart', e=>{ touchStartX=e.touches[0].clientX; touchStartY=e.touches[0].clientY; });
    document.addEventListener('touchend', e=>{
        let dx=e.changedTouches[0].clientX-touchStartX;
        let dy=e.changedTouches[0].clientY-touchStartY;
        if(Math.abs(dx)>Math.abs(dy)){
            if(dx>0) playerX+=40; else playerX-=40;
        } else { if(dy>0) playerY+=40; else playerY-=40; }
        updatePlayer();
    });
    gameInterval=setInterval(checkCollision,100);
    setTimeout(showLetter,12000);
}
let playerX=window.innerWidth/2, playerY=50;
function movePlayer(e){
    const step=20;
    if(e.key==='ArrowLeft') playerX-=step;
    if(e.key==='ArrowRight') playerX+=step;
    if(e.key==='ArrowUp') playerY-=step;
    if(e.key==='ArrowDown') playerY+=step;
    updatePlayer();
}
function updatePlayer(){
    if(playerX<0) playerX=0;
    if(playerX>window.innerWidth-50) playerX=window.innerWidth-50;
    if(playerY<0) playerY=0;
    if(playerY>game.offsetHeight-50) playerY=game.offsetHeight-50;
    player.style.left=playerX+'px';
    player.style.top=playerY+'px';
}
function checkCollision(){
    salamis.forEach((s,index)=>{
        let sX=parseFloat(s.style.left);
        let sY=parseFloat(s.style.top);
        if(Math.abs(playerX-sX)<40 && Math.abs(playerY-sY)<40){
            score++;
            scoreBoard.innerText='Score: '+score;
            s.style.transform='scale(1.5)';
            setTimeout(()=>{if(s) s.remove();},200);
            salamis.splice(index,1);
        }
    });
}

// Show letter with animation & confetti
function showLetter(){
    clearInterval(gameInterval);
    document.getElementById('game').classList.add('hidden');
    const letter=document.getElementById('letter');
    letter.classList.remove('hidden');
    // Fade-in lines
    let lines=document.querySelectorAll('#letter p');
    lines.forEach((line,i)=>{
        setTimeout(()=>{line.style.opacity=1; line.style.animation='fadeInLine 1s forwards';}, i*800);
    });
    // Show GIF
    setTimeout(()=>{document.getElementById('letterGif').style.display='inline'; createConfetti();}, lines.length*800);
}

// Confetti
function createConfetti(){
    for(let i=0;i<50;i++){
        let c=document.createElement('div');
        c.classList.add('confetti-piece');
        c.style.left=Math.random()*window.innerWidth+'px';
        c.style.background=['#ffd700','#ff6b6b','#1abc9c','#f39c12'][Math.floor(Math.random()*4)];
        c.style.animationDuration=(Math.random()*2+2)+'s';
        document.body.appendChild(c);
        setTimeout(()=>{c.remove();},4000);
    }
}
